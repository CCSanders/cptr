export const listUserFollowers = /* GraphQL */ `
    query ListUserFollowers($userId: ID!) {
        getUser(id: $userId) {
            followers {
                items {
                    followerId
                }
            }
        }
    }

export const followUser = /* GraphQL */ `
    mutation FollowUser($followeeId: ID!, $followerId: ID!) {
        createFollowRelationship(
            input: {
                followerId: $followerId, 
                followeeId: $followeeId, 
                userFollowersId: $followeeId, 
                userFollowingId: $followerId
        }) {
            followeeId
            followerId
            userFollowersId
            userFollowingId
        }
    }
`;
